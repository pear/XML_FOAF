71        /**
72         * Create new FOAF Agent
73         *
74         * @foafstatus Unstable
75         * @param string $agent_type Agent type, this can be Person, Organization, Group, Agent.
76         * @access public
77         * @return void 
78         * @link http://xmlns.com/foaf/0.1/#term_Organization FOAF Specification - foaf:Organization
79         * @link http://xmlns.com/foaf/0.1/#term_Group FOAF Specification - foaf:Group
80         * @link http://xmlns.com/foaf/0.1/#term_Person FOAF Specification - foaf:Person
81         * @link http://xmlns.com/foaf/0.1/#term_Agent FOAF Specification - foaf:Agent
82         */
83    
84        function newAgent($agent_type = 'Person')
85        {
86            require_once 'XML/Tree.php';
87            $this->xml_tree = new XML_Tree;
88            $agent_type = strtolower($agent_type);
89            $this->agent = $agent_type;
90            switch ($agent_type) {
91                case 'group':
92                    $this->foaf =& $this->xml_tree->addRoot('foaf:Group');
93                    break;
94                case 'organization':
95                    $this->foaf =& $this->xml_tree->addRoot('foaf:Organization');
96                    break;
97                case 'agent':
98                    $this->foaf =& $this->xml_tree->addRoot('foaf:Agent');
99                case 'person':
100               default:
101                   $this->foaf =& $this->xml_tree->addRoot('foaf:Person');
102                   break;
103           }
104       }

271        * Add a foaf:mbox or foaf:mbox_sha1sum element
272        *
273        * @foafstatus testing
274        * @param string $mbox Mailbox, either a mailto:addr, addr or an sha1 sum of mailto:addr
275        * @param boolean $sha1 Whether or not to use foaf:mbox_sha1sum
276        * @param boolean $is_sha1_hash Whether or not given $mbox is already an sha1 sum
277        * @access public
278        * @return void 
279        * @see XML_FOAF::setMboxSha1Sum
280        * @link http://xmlns.com/foaf/0.1/#term_mbox_sha1sum FOAF Specification - foaf:mbox_sha1sum
281        * @link http://xmlns.com/foaf/0.1/#term_mbox FOAF Specification - foaf:mbox
282        */
283   
284       function addMbox($mbox,$sha1 = false,$is_sha1_hash = false)
285       {
286           if (substr($mbox,0,7) != 'mailto:' && $is_sha1_hash == false) {
287               $mbox = 'mailto:' .$mbox;
288           }
289   
290           if ($sha1 == true) {
291               if ($is_sha1_hash == false) {
292                   $mbox = sha1($mbox);
293               }
294               $this->children['mbox_sha1sum'][] =& $this->foaf->addChild('foaf:mbox_sha1sum', $mbox);
295           } else {
296               $this->children['mbox'][] =& $this->foaf->addChild('foaf:mbox', '', array('rdf:resource' => $mbox));
297           }
298   
299       }

424       function addOnlineAccount($account_name,$account_service_homepage = null,$account_type = null)
425       {
426           $holds_account =& $this->foaf->addChild('foaf:holdsAccount');
427           $online_account =& $holds_account->addChild('foaf:OnlineAccount');
428           $online_account->addChild('foaf:accountName', $account_name);
429           if (!is_null($account_service_homepage)) {
430               $online_account->addChild('foaf:accountServiceHomepage', '', array('rdf:resource' => $account_service_homepage));
431           }
432           if (!is_null($account_type)) {
433               $online_account->addChild('rdf:type', '', array('rdf:resource' => $account_type));
434           }
435           $this->children['holdsaccount'][] =& $holds_account;
436       }
