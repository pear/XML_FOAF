170       /**
171   	 * Parses our the foaf:Agents
172   	 *
173   	 * Looks for all foaf:Agents (foaf:Person,foaf:Group,foaf:Organzation and foaf:Agent)
174   	 * and decides which are the primary agents (who/what the FOAF is about) and
175   	 * which are only known by the primary agents
176   	 *
177   	 * @access private
178   	 * @return void 
179   	 */
180       
181       function _fetchAgent()
182       {
183           $person_resource = new Resource(XML_FOAF_NS . 'Person');
184           $persons = $this->foaf->find(null,null,$person_resource);
185           $group_resource = new Resource(XML_FOAF_NS . 'Group');
186           $groups = $this->foaf->find(null,null,$group_resource);
187           $organization_resource = new Resource(XML_FOAF_NS . 'Organization');
188           $organizations = $this->foaf->find(null,null,$organization_resource);
189           $agent_resource = new Resource(XML_FOAF_NS . 'Agent');
190           $agents = $this->foaf->find(null,null,$agent_resource);
191           $agents->addModel($persons);
192           $agents->addModel($groups);
193           $agents->addModel($organizations);
194           $knows_resource = new Resource(XML_FOAF_NS . 'knows');
195           $knows = $this->foaf->find(null,$knows_resource,null);
196           $i = 0;
197           $agent_nodes = array();
198           $known_nodes = array();
199           foreach ($agents->triples as $agent) {
200               $agent_nodes[$agent->subj->uri] = $agent->obj->uri;
201               $i += 1;
202               foreach ($knows->triples as $know) {
203                   if ($agent->subj->uri == $know->obj->uri) {
204                       $agent_type = pathinfo($agent->obj->uri);
205                       $agent_type = $agent_type['basename'];
206                       $known = array('node' => $know->obj->uri, 'agent' => $agent_type);
207                       $known_nodes["{$know->subj->uri}"][] = $known;
208                       $this->known_nodes[] = $know->obj->uri;
209                       unset($agent_nodes[$agent->subj->uri]);
210                   }
211               }
212           }
213           
214           $agents = array();
215           $i = 0;
216           foreach ($agent_nodes as $node => $agent_type) {
217               $agent_type = pathinfo($agent_type);
218               $agent_type = $agent_type['basename'];
219               $agents[$i] = array ('node' => $node, 'agent' => $agent_type);
220               $this->agent_nodes[] = $node;
221               if (isset($known_nodes[$node])) {
222                   foreach ($known_nodes[$node] as $knows) {
223                       $agents[$i]['knows'][] = $knows;
224                   }
225               }
226               $i += 1;                
227           }
228           return $agents;
229       }

238       function _getName()
239       {
240           $name_resource = new Resource(XML_FOAF_NS . 'name');
241           $names = $this->foaf->find(null,$name_resource,null);
242           foreach ($names->triples as $name) {
243               if (in_array($name->subj->uri,$this->agent_nodes)) {
244                   foreach ($this->foaf_data as $key => $agent) {
245                       if (isset($agent['node']) && ($agent['node'] == $name->subj->uri)) {
246                           $this->foaf_data[$key]['name'] = $name->obj->label;
247                           break;
248                       }
249                   }
250               } elseif (in_array($name->subj->uri,$this->known_nodes)) {
251                   foreach ($this->foaf_data as $agent_key => $agent) {
252                       foreach ($agent['knows'] as $knows_key => $knows) {
253                           if (isset($knows['node']) && ($knows['node'] == $name->subj->uri)) {
254                               $this->foaf_data[$agent_key]['knows'][$knows_key]['name'] = $name->obj->label;
255                               break 2;
256                           }
257                       }
258                   }
259               }
260           }
261       }

788       /**
789        * Return parsed FOAF data as an Object
790        *
791        * @access public
792        * @return object 
793        */
794       
795       function toObject()
796       {
797           $foaf_object = new stdClass();
